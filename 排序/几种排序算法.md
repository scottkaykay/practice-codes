## 几种排序算法(c++ version)

## 1. 冒泡排序  平均时间复杂度O(N^2), 最好O(N),最坏O(N^2)  空间复杂度O(1)，稳定：排序后相同键值的顺序和排序前相同

```C++
#include<iostream>
#include<vector>
using namespace std;

void swap(int& a,int& b)
{
    int temp=a;
    a=b;
    b=temp;
}

int* bubblesort(int arr[],int length)
{
    if(length==1)
        return arr;
    for(int i=0;i<length;i++)
    {
        for(int j=0;j<length-i-1;j++)
        {
            if(arr[j]>arr[j+1])
                swap(arr[j],arr[j+1]);
        }
    }
    return arr;
}
```

## 2. 选择排序  平均时间复杂度O(N^2),最好，坏时间复杂度O(N^2)，空间复杂度O(1)，不稳定

```C++

void selectosort(int arr[],int length)

        return;

    {
        int temp=i,minvalue=arr[i];
        for(int j=i+1;j<length;j++)

            {
                minvalue=arr[j];

            }
        }
        swap(arr[i],arr[temp]);
    }
```

## 3.插入排序 平均时间复杂度O(N^2) 最佳O(N), 最坏O(N^2) ，空间复杂度O(1) , 稳定

```C++
#include<iostream>
#include<vector>
using namespace std;

void insertsort(int arr[],int length)
{
    if(length==1)
        return;
    for(int i=1;i<length;i++)
    {
        int j=0;
        for(j=i-1;j>=0 && arr[j+1]<arr[[j];j--)
        {
            swap(arr[j+1],arr[j]);
        }
        if(j==i-1)
            arr[j+1]=arr[i];
    }
}
```

## 4.希尔排序 平均时间复杂度O(NlogN) 空间复杂度O(1)， 不稳定

```C++
#include<iostream>

void shellsort(int arr[],int length)
{
    if(length==1)
        return;
    int gap=length/2;
    int temp=0;
    while(gap>0)
    {
        for(int i=gap;i<length;i++)
        {
            temp=arr[i];
            int pre=i-gap;
            while(pre>=0 && arr[pre]>temp)
            {
                arr[pre+gap]=arr[pre];
                pre-=gap;
             }
        arr[pre+gap]=temp;
        }
        gap/=2;
    }
}
```

## 5. 归并排序 平均时间复杂度O(NlogN)  空间复杂度：O（N） 稳定

```C++
#include<iostream>
#include<vector>
using namespace std;

vector<int> Merge(vector<int>& left,vector<int>& right)
{
    int lenleft=left.size();
    int lenright=right.size();
    vector<int> res(lenleft+lenright,0);
    for(vector<int>::size_type i=0,j=0,index=0;index<lenleft+lenright;index++)
    {
        if(i>=lenleft)
        {
            res[index]=right[j++];
        }
        else if(j>=lenright)
        {
            res[index]=left[i++];
        }
        else
        {
            if(left[i]<right[j])
            {   res[index]=left[i];
                i++;
                }
            else
            {
                res[index]=right[j];
                j++;
            }
        }
        
        
    }
    return res;
}

vector<int> MergeSort(vector<int>& arr)
{
    int length=arr.size();
    if(length==1)
        return  arr;
    int index=length/2;
    vector<int> left(index,0);
    vector<int> right(length-index,0);
    vector<int>::size_type iter=0;
    for(vector<int>::size_type l=left.begin();l!=left.end();l++)
    {
        *l=arr[iter++];
    }
    for(vector<int>::size_type r=right.begin();r!=right.end();r++)
    {
        *r=arr[iter++];
    }
    return Merge(MergeSort(left),MergeSort(right));
}
```

## 8. 计数排序 平均时间复杂度：O(n+k) 最佳，最差：O(n+k) 空间复杂度：O(k) k为数据量 稳定

```C++
void Countingsort(int arr[], int length)
{
	//只能对整数排序
	
	if (length < 1)
		return;
	int min = arr[0], max = arr[0];
	for (int i = 0; i < length; i++)
	{
		if (arr[i] > max)
			max = arr[i];
		if (arr[i] < min)
			min = arr[i];
	}
	int bias = 0 - min;
	vector<int> bucket(max - min + 1, 0);
	for (int i = 0; i < length; i++)
		bucket[arr[i] + bias]++;
	int index = 0, i = 0;
	while (index < length)
	{
		if (bucket[i] != 0)
		{
			arr[index] = i - bias;
			index++;
			bucket[i]--;
		}
		else
			i++;
	}

}

```

## 9. 桶排序 平均时间复杂度O（n+k)  最佳：O(n+k) 最坏：O(n^2) 空间复杂度：O(n+k) 稳定

```C++
vector<int> Bucketsort(vector<int> arr,int bucketsize)
{
	
	int length = arr.size();
	if (length <= 1)
		return arr;
	//找到最大值最小值
	int max = arr[0], min = arr[0];
	for (vector<int>::size_type i = 0; i < length; i++)
	{
		if (arr[i] > max)
			max = arr[i];
		if (arr[i] < min)
			min = arr[i];
	}
	int bucketcount = (max - min) / bucketsize + 1; //桶的数量
	vector<vector<int>> bucket(bucketcount);
	vector<int> res;
	//把元素放到对应的桶里
	for (vector<int>::size_type i = 0; i < length; i++)
	{
		bucket[(arr[i] - min) / bucketsize].push_back(arr[i]);//如0~10的数，bucketsize=5,0放在0号桶，5放在1号桶，10放在32号桶
	}
	for (vector<int>::size_type i = 0; i < bucketcount; i++)
	{
		if (bucketsize == 1)//1个桶一种元素
		{
			for (vector<int>::size_type j = 0; j < bucket[i].size(); j++)
				res.push_back(bucket[i][j]);

		}
		else
		{
			if (bucketcount == 1)//只有一个桶
				bucketsize--;
			vector<int> temp = Bucketsort(bucket[i], bucketsize);//递归桶排序
			for (vector<int>::size_type i = 0; i < temp.size(); i++)
				res.push_back(temp[i]);

		}
	}
	return res;
}

```

## 10 基数排序 平均时间复杂度：O(n*k) 最佳，最差：O(n*k) 

```C++
void Radixsort(int arr[],int length)
{
	if (arr == NULL)
		return ;
	
	if (length == 1)
		return;
	//计算最大数位数，决定遍历几次
	int max = arr[0];
	for (int i = 1; i < length; i++)
	{
		if (arr[i] > max)
			max = arr[i];
	}
	int maxdigit = 0;
	while (max != 0)
	{
		max /= 10;
		maxdigit++;
	}

	int mod = 10,div = 1;
	vector<vector<int>> bucket(10);
	for (int i = 0; i < maxdigit; i++, mod *= 10, div *= 10)
	{
		for (int j = 0; j < length; j++)
		{
			int num = arr[j] % mod / div;
			bucket[num].push_back(arr[j]);
		}
		//把元素全部从容器拿出来，进行下一位的排序
		int index = 0;
		for (vector<int>::size_type j = 0; j < bucket.size(); j++)
		{
			for (vector<int>::size_type k = 0; k < bucket[j].size(); k++)
			{
				arr[index++] = bucket[j][k];
			}
		}
		bucket.clear();

	}
}
```
