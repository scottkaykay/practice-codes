### 给出一个已知长度的字符串，找出只包含两种字符的最长连续子串

输入：\

8

1 2 2 5 6 9 8 7

答案可能有多个，输出其中的一种： 1 2 2

```C++

#include<iostream>
#include<vector>
#include<queue>

{
	int n, val;
	vector<int> store;

	{
		cin >> val;
		store.push_back(val);
	}
	int maxvalue = 0;
	string s = "";
	for (int v : store)
		s += (v + '0');
	if (n <= 2)
	{
		cout << s << endl;
		return 0;
	}
	int start = 0, end = 0;
	int i = 0, j = 0;
	
	char duilie[2]={ '*','*' };

	while (end < n)
	{
		if (duilie[0] == '*')
			duilie[0] = s[end];
		else if (duilie[1] == '*')
		{
			if(duilie[0]!=s[end])
				duilie[1] = s[end];
		}
		else
		{
			if (s[end] != duilie[0] && s[end] != duilie[1])
			{
				if (end - start > maxvalue)
				{
					i = start;
					j = end;
					maxvalue = j - i;
				}
				start = end;
			}
		}
		++end;
	}

	return 0;
```
