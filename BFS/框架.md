```C++
int BFS(Node start,Node target)
{
    queue<Node> q;
    set<Node> visited;//防止走回头路
    
    q.push(start);
    visited.add(start);
    int step=0;//记录扩散的步数
    
    while(!q.empty())
    {
        int size=q.size();
        while(size-->0)
        {
            Node cur=q.front();
            q.pop();
            if(cur==target)
                return step;
            for(Node x:cur.adj())
            {
                if(x not in visited)
                {
                    q.push(x);
                    visited.add(x);
                }
            }
        }
        step++;
    }
   //cur.adj() 泛指 cur 相邻的节点，比如说二维数组中，cur 上下左右四面的位置就是相邻节点；visited 的主要作用是防止走回头路，大部分时候都是必须的，但是像一般的二叉树结构，没有子节点到父节点的指针，不会走回头路就不需要 visited
}
```
