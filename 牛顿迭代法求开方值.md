## 牛顿迭代法求开方值

## 代码

```C++
// x^2=a
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int a;
    cin>>a;
    double x=a;
    double last=a;
    while(true)
    {
        last=x;
        x=last-(last*last-a)/(2*last);
        if(abs(x-last)<eps)
            break;
    }
    cout<<fixed<<setprecision(1)<<x<<endl;
    return 0;
}

//x^3=a

int main()
{
    int a;
    cin>>a;
    double x=a;
    double last=a;
    while(true)
    {
        last=x;
        x=last-(last*last*last-a)/(3*last*last);
        if(abs(x-last)<eps)
            break;
    }
    cout<<fixed<<setprecision(1)<<x<<endl;
    return 0;
}
```
